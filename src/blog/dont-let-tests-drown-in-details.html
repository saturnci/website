Don't Let Tests Drown in Details

Recently I asked Claude to write tests for

context "when the user was created within the last 24 hours" do
  it "highlights the Created cell with #0F0" do
    create(:user, name: "Just Created", created_at: 1.hour.ago)
    get admin_users_path

    formatted = I18n.l(1.hour.ago.in_time_zone(Rails.application.config.admin_time_zone))
    expect(response.body).to match(%r{]*style="[^"]*background-color: ?#0F0[^"]*"[^>]*>\s*#{Regexp.escape(formatted)}}i)
  end
end
